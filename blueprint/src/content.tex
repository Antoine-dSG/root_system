\chapter{Root pairings}

\section{Definitions}

Recall that a \emph{perfect pairing} is a quadruple $(R,M,N,\mathcal{L})$ where
$R$ is a commutative ring, $M$ and $N$ are $R$-modules, and
$\mathcal{L}:M\times N\to R$ is a bilinear map such that the induced maps
$M\to \operatorname{Hom}_R(N,R)$ and $N\to \operatorname{Hom}_R(M,R)$ are
isomorphisms of $R$-modules. If $I$ is a set, denote by
$\operatorname{Perm}(I)$ the group of permutations of $I$.

\begin{definition}\label{def:root_pairing}
\lean{RootPairing}
\leanok
A \emph{root pairing} is a tuple
$(R,M,N,\mathcal{L},I,\alpha,\alpha^\vee,s)$ where:
\begin{itemize}
    \item $(R,M,N,\mathcal{L})$ is a perfect pairing,
    \item $I$ is a set,
    \item The maps 
    \begin{align*}
        \alpha:&\; I\to M, \quad i\mapsto \alpha_i, \\
        \alpha^\vee:&\; I\to N, \quad i\mapsto \alpha_i^\vee, \\
        s:&\; I\to \operatorname{Perm}(I), \quad i\mapsto s_i,
    \end{align*}
    are such that
    \begin{enumerate}
    \item $\alpha$ and $\alpha^\vee$ are injective,
    \item for all $i\in I$, $\mathcal{L}(\alpha_i,\alpha_i^\vee)=2$,
    \item for all $i,j\in I$,
    \begin{align*}
        \alpha_{s_i(j)}
            &=\alpha_j-\mathcal{L}(\alpha_j,\alpha_i^\vee)\alpha_i,\\
        \alpha_{s_i(j)}^\vee
            &=\alpha_j^\vee-\mathcal{L}(\alpha_i,\alpha_j^\vee)\alpha_i^\vee.
    \end{align*}
\end{enumerate} 
\end{itemize}

\end{definition}

\begin{definition}\label{def:root_pairing_properties}
\lean{RootPairing.IsCrystallographic, RootPairing.IsReduced, Finite}
\uses{def:root_pairing}
\leanok
Fix a root pairing $\Phi=(R,M,N,\mathcal{L},I,\alpha,\alpha^\vee,s)$.
We say that $\Phi$ is:
\begin{itemize}
    \item \emph{crystallographic} if
    $\mathcal{L}(\alpha_i,\alpha_j^\vee)\in \mathbb{Z}$ for all $i,j\in I$,
    \item \emph{reduced} if, for all $i,j\in I$, the roots $\alpha_i$ and
    $\alpha_j$ are linearly dependent over $R$ if and only if
    $\alpha_i=\pm\alpha_j$,
    \item \emph{finite} if $I$ is finite.
\end{itemize}
\end{definition}


\chapter{Constructions}\label{chapter:constructions}

\section{Type A}

Consider the perfect pairing
\[
    (R,M,N,\mathcal{L})
    =
    \bigl(\mathbb{Z},\mathbb{Z}^n,(\mathbb{Z}^n)^*,
    \langle\cdot,\cdot\rangle\bigr),
\]
where $\langle\cdot,\cdot\rangle$ is the canonical pairing between
$\mathbb{Z}^n$ and its algebraic dual
$(\mathbb{Z}^n)^*=\operatorname{Hom}_{\mathbb{Z}}(\mathbb{Z}^n,\mathbb{Z})$.
Let $e_1,\ldots,e_n$ be the standard basis of $\mathbb{Z}^n$, and let
$e_1^*,\ldots,e_n^*$ be the dual basis.

Write $[1,n]=\{1,\ldots,n\}$. An \emph{interval} in $[1,n]$ is a set
$[i,j]=\{i,i+1,\ldots,j\}$ with $1\le i\le j\le n$. We denote by $I_n$
the set of intervals in $[1,n]$ and write $[i]=[i,i]$.

A \emph{signed interval} in $[1,n]$ is an element of
$I_n\times\{\pm1\}$. We identify $J\in I_n$ with $(J,1)$ and write
$-J$ for $(J,-1)$.

Let $A=(a_{u,r})$ be the $n\times n$ integer matrix defined by
\[
    a_{u,r}=
    \begin{cases}
        2 & \text{if }u=r,\\
        -1 & \text{if }|u-r|=1,\\
        0 & \text{otherwise.}
    \end{cases}
\]
This is the Cartan matrix of type $A_n$.

Define
\begin{align}
    \alpha &: I_n\times\{\pm1\}\to\mathbb{Z}^n,
    &
    \alpha_{\pm J} &= \pm\sum_{r\in J}Ae_r,
    \label{eq:alpha_definition}\\
    \alpha^\vee &: I_n\times\{\pm1\}\to(\mathbb{Z}^n)^*,
    &
    \alpha_{\pm J}^\vee &= \pm\sum_{r\in J}e_r^*.
    \label{eq:alpha_dual_definition}
\end{align}

\begin{lemma}\label{lemma:pairing_formula}
\lean{pairing_formula}
\uses{def:root_pairing}
\leanok
Let $J=\varepsilon[i,j]$ and $K=\delta[k,l]$ be signed intervals. Then
\[
    \mathcal{L}(\alpha_K,\alpha_J^\vee)
    =
    \varepsilon\delta
    \bigl(
        \delta_{i,k}
        +\delta_{j,l}
        -\delta_{i,l+1}
        -\delta_{k,j+1}
    \bigr).
\]
\end{lemma}
\begin{proof}
    \leanok
It is enough to consider $J,K \in I_n$, since the signs factor out by
bilinearity. For fixed $r$, the vector $Ae_r$ has coefficient $2$ in
coordinate $r$, coefficient $-1$ in the adjacent coordinates $r-1$ and
$r+1$, and coefficient $0$ elsewhere. Hence, for $u\in[1,n]$,
\[
    \sum_{r=k}^{l} a_{u,r}
    =
    \delta_{u,k}
    +\delta_{u,l}
    -\delta_{u,k-1}
    -\delta_{u,l+1},
\]
where $\delta_{x,y}$ is the Kronecker delta. Summing over $u=i,\ldots,j$ gives the displayed formula.
\end{proof}

\begin{lemma}\label{lemma:alpha_dual_alpha_is_two}
\lean{pairing_self, root_coroot_two'}
\uses{lemma:pairing_formula}
\leanok
For every $J\in I_n\times\{\pm1\}$,
$\mathcal{L}(\alpha_J,\alpha_J^\vee)=2$.
\end{lemma}
\begin{proof}
    \leanok
Write $J=\varepsilon[i,j]$. In Lemma \ref{lemma:pairing_formula}, the two
positive endpoint terms are both $1$, while the two negative endpoint
conditions $i=j+1$ are impossible because $i\le j$. Thus
\[
    \mathcal{L}(\alpha_J,\alpha_J^\vee)
    =
    \varepsilon^2(1+1)=2.
\]
\end{proof}

\begin{lemma}\label{lemma:detail_pairing_eq_two}
\lean{pairing_eq_two_iff_eq}
\uses{lemma:pairing_formula,lemma:alpha_dual_alpha_is_two}
\leanok
For signed intervals $J$ and $K$,
\[
    \mathcal{L}(\alpha_K,\alpha_J^\vee)=2
    \quad\Longleftrightarrow\quad
    K=J.
\]
\end{lemma}
\begin{proof}
    \leanok
Write $J=\varepsilon[i,j]$ and $K=\delta[k,l]$. Let
\[
    E=
    \delta_{i,k}
    +\delta_{j,l}
    -\delta_{i,l+1}
    -\delta_{k,j+1}.
\]
By Lemma \ref{lemma:pairing_formula},
$\mathcal{L}(\alpha_K,\alpha_J^\vee)=\varepsilon\delta E$.
The two negative endpoint conditions cannot both hold: if
$i=l+1$ and $k=j+1$, then $l<i\le j<k\le l$, a contradiction. Hence
$E\ge -1$. Since also $E\le2$, the equality $\varepsilon\delta E=2$
forces $\varepsilon\delta=1$ and $E=2$.

The equality $E=2$ forces both positive endpoint terms to be $1$, so
$i=k$ and $j=l$. Since $\varepsilon\delta=1$ and
$\varepsilon,\delta\in\{\pm1\}$, we also have $\varepsilon=\delta$.
Thus $K=J$. The converse is Lemma \ref{lemma:alpha_dual_alpha_is_two}.
\end{proof}

\begin{lemma}\label{lemma:alpha_and_alpha_dual_injective}
\lean{α_injective, α_dual_injective}
\uses{lemma:detail_pairing_eq_two}
\leanok
The maps
$\alpha:I_n\times\{\pm1\}\to\mathbb{Z}^n$ and
$\alpha^\vee:I_n\times\{\pm1\}\to(\mathbb{Z}^n)^*$ are injective.
\end{lemma}
\begin{proof}
    \leanok
Suppose first that $\alpha_J=\alpha_K$. Pairing both sides with
$\alpha_J^\vee$ gives
\[
    2=\mathcal{L}(\alpha_J,\alpha_J^\vee)
    =\mathcal{L}(\alpha_K,\alpha_J^\vee).
\]
Lemma \ref{lemma:detail_pairing_eq_two} gives $K=J$.

Now suppose that $\alpha_J^\vee=\alpha_K^\vee$. Pairing both sides with
$\alpha_J$ gives
\[
    2=\mathcal{L}(\alpha_J,\alpha_J^\vee)
    =\mathcal{L}(\alpha_J,\alpha_K^\vee).
\]
Applying Lemma \ref{lemma:detail_pairing_eq_two} with the roles of the
two signed intervals exchanged gives $J=K$.
\end{proof}

We first define $s_J(K)$ for $J,K\in I_n$. Set
$L=(J\cup K)\setminus(J\cap K)$. Exactly one of the following cases occurs:
\begin{description}
    \item[{\bf R1.}] $L=\emptyset$.
    \item[{\bf R2.}] $L\in I_n$ and $K\subset J$.
    \item[{\bf R3.}] $L\in I_n$ and $K\not\subseteq J$.
    \item[{\bf R4.}] $L\notin I_n\cup\{\emptyset\}$.
\end{description}
Then
\begin{equation}\label{eq:s_J_definition}
s_J(K)=
\begin{cases}
    -K & \text{in case }{\bf R1},\\
    -L & \text{in case }{\bf R2},\\
    L & \text{in case }{\bf R3},\\
    K & \text{in case }{\bf R4}.
\end{cases}
\end{equation}

If $J=[i,j]$ and $K=[k,l]$, this is equivalently
\begin{equation}\label{eq:s_J_interval_definition}
s_{[i,j]}[k,l]=
\begin{cases}
    -[k,l] & \text{if }(i,j)=(k,l),\\
    {-[l+1,j]} & \text{if }i=k,\ j>l,\\
    {[j+1,l]} & \text{if }i=k,\ j<l,\\
    {-[i,k-1]} & \text{if }i<k,\ j=l,\\
    {[k,i-1]} & \text{if }i>k,\ j=l,\\
    {[k,j]} & \text{if }i=l+1,\\
    {[i,l]} & \text{if }k=j+1,\\
    {[k,l]} & \text{otherwise.}
\end{cases}
\end{equation}

Extend $s_J$ to signed intervals by $s_J(-K)=-s_J(K)$, and extend the
first argument by $s_{-J}=s_J$.

\begin{lemma}\label{lemma:s_J_are_permutations}
\lean{s, s_involutive}
\uses{def:root_pairing}
\leanok
For all $J\in I_n\times\{\pm1\}$, we have
$s_J\in\operatorname{Perm}(I_n\times\{\pm1\})$.
\end{lemma}
\begin{proof}
    \leanok
A direct case check from (\ref{eq:s_J_interval_definition}) shows that
$s_J(s_J(K))=K$ for every signed interval $K$. Hence $s_J$ is its own
inverse, and therefore a permutation.
\end{proof}

\begin{lemma}\label{lemma:s_is_refl_on_alpha}
\lean{reflectionPerm_root_eval, reflectionPerm_root'}
\uses{lemma:pairing_formula}
\leanok
For all $J,K\in I_n\times\{\pm1\}$,
\[
    \alpha_{s_J(K)}
    =
    \alpha_K-\mathcal{L}(\alpha_K,\alpha_J^\vee)\alpha_J.
\]
\end{lemma}
\begin{proof}
First assume that $J=[i,j]$ and $K=[k,l]$ are positive intervals. Put
$c=\mathcal{L}(\alpha_K,\alpha_J^\vee)$. Lemma
\ref{lemma:pairing_formula} gives
\[
    c=
    \delta_{i,k}
    +\delta_{j,l}
    -\delta_{i,l+1}
    -\delta_{k,j+1}.
\]
Comparing this value with (\ref{eq:s_J_interval_definition}) gives:
\[
\begin{array}{c|c|c}
\text{condition} & c & \alpha_K-c\alpha_J \\ \hline
(i,j)=(k,l) & 2 & -\alpha_{[k,l]}\\
i=k,\ j>l & 1 & -\alpha_{[l+1,j]}\\
i=k,\ j<l & 1 & \alpha_{[j+1,l]}\\
i<k,\ j=l & 1 & -\alpha_{[i,k-1]}\\
i>k,\ j=l & 1 & \alpha_{[k,i-1]}\\
i=l+1 & -1 & \alpha_{[k,j]}\\
k=j+1 & -1 & \alpha_{[i,l]}\\
\text{otherwise} & 0 & \alpha_{[k,l]}.
\end{array}
\]
Each entry in the last column is precisely $\alpha_{s_J(K)}$.

For arbitrary signs, write $J=\varepsilon J_0$ and $K=\delta K_0$ with
$J_0,K_0\in I_n$ and $\varepsilon,\delta\in\{\pm1\}$. Since
$s_{\varepsilon J_0}(\delta K_0)=\delta s_{J_0}(K_0)$ and
\[
    \mathcal{L}(\alpha_{\delta K_0},\alpha_{\varepsilon J_0}^\vee)
    =
    \delta\varepsilon
    \mathcal{L}(\alpha_{K_0},\alpha_{J_0}^\vee),
\]
the positive case, multiplied by $\delta$, proves the signed case.
\end{proof}

\begin{lemma}\label{lemma:s_is_refl_on_alpha_dual}
\lean{pairing_symm, reflectionPerm_coroot_single, reflectionPerm_coroot'}
\uses{lemma:pairing_formula,lemma:s_is_refl_on_alpha}
\leanok
For all $J,K\in I_n\times\{\pm1\}$,
\[
    \alpha_{s_J(K)}^\vee
    =
    \alpha_K^\vee-\mathcal{L}(\alpha_J,\alpha_K^\vee)\alpha_J^\vee.
\]
\end{lemma}
\begin{proof}
    \leanok
For positive intervals, symmetry of the type $A$ Cartan matrix gives the
same endpoint value for $\mathcal{L}(\alpha_J,\alpha_K^\vee)$ as in the
proof of Lemma \ref{lemma:s_is_refl_on_alpha}. The case table there used
only additivity of sums over adjacent or nested intervals. Replacing each
$Ae_r$ by $e_r^*$ in that table gives the desired coroot identity for
positive intervals. The signed case follows by the same sign reduction as
in Lemma \ref{lemma:s_is_refl_on_alpha}.
\end{proof}

\begin{theorem}\label{thm:root_pairing_of_type_A}
\lean{An}
\uses{def:root_pairing,lemma:alpha_and_alpha_dual_injective,lemma:alpha_dual_alpha_is_two,lemma:s_J_are_permutations,lemma:s_is_refl_on_alpha,lemma:s_is_refl_on_alpha_dual,An_is_finite, An_is_crystallographic,An_is_reduced}
\leanok
Fix $n\in\mathbb{Z}_{>0}$, and let
\[
    R=\mathbb{Z},\qquad M=\mathbb{Z}^n,\qquad N=(\mathbb{Z}^n)^*.
\]
Let $\mathcal{L}$ be the canonical pairing, let
$I=I_n\times\{\pm1\}$ be the set of signed intervals, let $\alpha$ and
$\alpha^\vee$ be as in (\ref{eq:alpha_definition}) and
(\ref{eq:alpha_dual_definition}), and let $s$ be as in
(\ref{eq:s_J_definition}). Then
\[
    (R,M,N,\mathcal{L},I,\alpha,\alpha^\vee,s)
\]
is a finite, crystallographic and reduced root pairing.
\end{theorem}
\begin{proof}
    \leanok
The quadruple
$(\mathbb{Z},\mathbb{Z}^n,(\mathbb{Z}^n)^*,\mathcal{L})$ is the standard
perfect pairing between a finite free $\mathbb{Z}$-module and its dual.
It remains to verify the root-pairing axioms from Definition
\ref{def:root_pairing}.

The maps $\alpha$ and $\alpha^\vee$ are injective by Lemma
\ref{lemma:alpha_and_alpha_dual_injective}. The identity
$\mathcal{L}(\alpha_J,\alpha_J^\vee)=2$ is Lemma
\ref{lemma:alpha_dual_alpha_is_two}. Lemma
\ref{lemma:s_J_are_permutations} gives $s_J\in\operatorname{Perm}(I)$ for
each $J$. Finally, Lemmas \ref{lemma:s_is_refl_on_alpha} and
\ref{lemma:s_is_refl_on_alpha_dual} give the two reflection formulas.
These are exactly the conditions in Definition \ref{def:root_pairing}. This proves 
the tuple is a root pairing. It is finite because $I$ is finite, crystallographic by 
Lemma \ref{lemma:pairing_formula}, and reduced by 
(\ref{eq:alpha_definition}) and (\ref{eq:alpha_dual_definition}).
\end{proof}

\begin{definition}\label{def:root_pairing_base}
\lean{RootPairing.Base}
\uses{def:root_pairing}
\leanok
Given a root pairing $\Phi=(R,M,N,\mathcal{L},I,\alpha,\alpha^\vee,s)$,
a set $\Delta\subset I$ is called a \emph{base} of $\Phi$ if:
\begin{enumerate}
    \item the sets $\{\alpha_i:i\in\Delta\}$ and
    $\{\alpha_i^\vee:i\in\Delta\}$ are $R$-linearly independent, and
    \item for every $x\in I$, there exist
    $k,m\in\mathbb{Z}_{>0}$,
    $a_1,\ldots,a_k,b_1,\ldots,b_m\in\Delta$, and
    $\varepsilon,\delta\in\{\pm1\}$ such that
    \[
        \alpha_x=\varepsilon\sum_{r=1}^k\alpha_{a_r}
        \quad\text{and}\quad
        \alpha_x^\vee=\delta\sum_{r=1}^m\alpha_{b_r}^\vee.
    \]
\end{enumerate}
\end{definition}

\begin{definition}\label{def:irreducible_base}
\lean{RootPairing.IsIrreducible}
\uses{def:root_pairing,def:root_pairing_base}
\leanok
A base $\Delta$ is said to be \emph{irreducible} if there exist
$k\in\mathbb{Z}_{>0}$ and $a_1,\ldots,a_k\in\Delta$ such that:
\begin{enumerate}
    \item $\{a_1,\ldots,a_k\}=\Delta$, and
    \item $\mathcal{L}(\alpha_{a_r},\alpha_{a_{r+1}}^\vee)\ne0$ for every
    $1\le r<k$.
\end{enumerate}
By definition, a root pairing admitting an irreducible base is called \emph{irreducible}.
\end{definition}

\begin{definition}\label{def:cartan_type}
\lean{CartanMatrix.A, CartanMatrix.B, CartanMatrix.C, CartanMatrix.D, CartanMatrix.E₆, CartanMatrix.E₇, CartanMatrix.E₈}
\uses{def:root_pairing_base,def:irreducible_base}
\leanok
Fix a finite, crystallographic, reduced and irreducible root pairing 
$\Phi=(R,M,N,\mathcal{L},I,\alpha,\alpha^\vee,s)$ and a
base $\Delta\subset I$. The pair $(\Phi,\Delta)$ is said to be of:
\begin{enumerate}
    \item type $A$ if there is a sequence $(i_1,\ldots,i_n)$, where
    $n=|\Delta|$, of distinct elements of $\Delta$ such that
    \[
        \mathcal{L}(\alpha_{i_r},\alpha_{i_{r+1}}^\vee)=-1
        \quad\text{for all }1\le r<n;
    \]
    \item type $B$ if there is a sequence $(i_1,\ldots,i_n)$ of distinct
    elements of $\Delta$ such that
    \[
        \mathcal{L}(\alpha_{i_r},\alpha_{i_{r+1}}^\vee)=-1
        \quad\text{for all }1\le r<n-1,
        \qquad
        \mathcal{L}(\alpha_{i_n},\alpha_{i_{n-1}}^\vee)=-2;
    \]
    \item type $C$ if there is a sequence $(i_1,\ldots,i_n)$ of distinct
    elements of $\Delta$ such that
    \[
        \mathcal{L}(\alpha_{i_r},\alpha_{i_{r+1}}^\vee)=-1
        \quad\text{for all }1\le r<n-1,
        \qquad
        \mathcal{L}(\alpha_{i_{n-1}},\alpha_{i_n}^\vee)=-2;
    \]
    \item type $D$ if there exist distinct elements $d_1,d_2,t\in\Delta$
    such that
    \[
        \mathcal{L}(\alpha_{d_a},\alpha_t^\vee)=-1
        \quad(a=1,2),
    \]
    and $\Delta\setminus\{d_1,d_2\}$ is of type $A$;
    \item type $E$ if there exist distinct elements $d_1,d_2,t\in\Delta$
    such that
    \[
        \mathcal{L}(\alpha_{d_1},\alpha_{d_2}^\vee)=-1,
        \qquad
        \mathcal{L}(\alpha_{d_2},\alpha_t^\vee)=-1,
    \]
    and $\Delta\setminus\{d_1,d_2\}$ is of type $A$.
\end{enumerate}
\end{definition}
